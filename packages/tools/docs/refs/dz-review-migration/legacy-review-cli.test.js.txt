function stripAnsi(text) {
  return text.replace(/\x1b\[[0-9;]*m/g, "");
}

function formatLocalIso(year, month, day, hour, minute, second) {
  const date = new Date(year, month - 1, day, hour, minute, second);
  const offsetMinutes = -date.getTimezoneOffset();
  const sign = offsetMinutes < 0 ? "-" : "+";
  const absoluteOffset = Math.abs(offsetMinutes);
  const offset = `${sign}${String(Math.floor(absoluteOffset / 60)).padStart(2, "0")}:${String(absoluteOffset % 60).padStart(2, "0")}`;
  return `${String(year).padStart(4, "0")}-${String(month).padStart(2, "0")}-${String(day).padStart(2, "0")}T`
    + `${String(hour).padStart(2, "0")}:${String(minute).padStart(2, "0")}:${String(second).padStart(2, "0")}${offset}`;
}

test("collects HTML and custom review conversations with line numbers", () => {
  const text = [
    "Intro",
    "<!--",
    "@agent HTML note",
    "-->",
    "Body {?? @me custom note ??} end",
  ].join("\n");

  const conversations = collectConversations(text);

  assert.deepEqual(
    conversations.map(({ lineStart, lineEnd, roles }) => ({ lineStart, lineEnd, roles })),
    [
      { lineStart: 2, lineEnd: 4, roles: ["agent"] },
      { lineStart: 5, lineEnd: 5, roles: ["me"] },
    ],
  );
});

test("conversation actions delete, toggle ok, and append replies", () => {
  const source = "{?? @agent note ??}";
  const [conversation] = collectConversations(source);

  const withOk = applyConversationAction(source, conversation, { kind: "toggle-ok" });
  assert.equal(withOk, "{?? @agent note @me ok ??}");

  const withoutOk = applyConversationAction(withOk, collectConversations(withOk)[0], { kind: "toggle-ok" });
  assert.equal(withoutOk, "{?? @agent note ??}");

  const withReply = applyConversationAction(source, conversation, {
    kind: "reply",
    body: "Vu, je garde.",
  });
  assert.equal(withReply, "{?? @agent note @me Vu, je garde. ??}");

  const deleted = applyConversationAction(source, conversation, { kind: "delete" });
  assert.equal(deleted, "");
});

test("detects closed conversations by trailing human ok replies", () => {
  assert.equal(isClosedConversation(collectConversations("<!-- @agent note @me ok -->")[0]), true);
  assert.equal(isClosedConversation(collectConversations("<!-- @agent note @me: OK -->")[0]), true);
  assert.equal(isClosedConversation(collectConversations("{?? @agent note @ ok ??}")[0]), true);
  assert.equal(isClosedConversation(collectConversations("<!-- @agent note @me question -->")[0]), false);
  assert.equal(isClosedConversation(collectConversations("<!-- @agent note @agent ok -->")[0]), false);
});

test("classifies conversation status from the trailing review message", () => {
  const statusOf = (text) => getConversationStatus(collectConversations(text)[0]);

  assert.equal(statusOf("<!-- @agent note -->"), "open");
  assert.equal(statusOf("<!-- @agent note @me -->"), "wip");
  assert.equal(statusOf("<!-- @agent note @ -->"), "wip");

  assert.equal(statusOf("<!-- @agent note @ answer -->"), "handled");
  assert.equal(statusOf("<!-- @agent note @me ok -->"), "resolved");

});

test("encodes and decodes compact review timestamps", () => {
  const instant = new Date("2026-06-16T17:35:35+02:00");

  assert.equal(encodeCompactTimestamp(instant, 120), "1WzvP91W");
  assert.deepEqual(decodeCompactTimestamp("1WzvP91W"), {
    offsetMinutes: 120,
    unixSeconds: 1781624135n,
  });
  assert.equal(
    formatTimestampForDisplay(parseReviewTimestamp("1WzvP91W")),
    "2026-06-16T17:35:35+02:00",
  );
  assert.equal(
    formatTimestampForDisplay(parseReviewTimestamp("%1WzvP91W")),
    "2026-06-16T17:35:35+02:00",
  );
  assert.equal(
    formatTimestampForDisplay(parseReviewTimestamp("2026-06-16T17:35:35+0200")),
    "2026-06-16T17:35:35+02:00",
  );
});

test("collects review annotations and applies or cancels them", () => {
  const source = [
    "Add {++%1WzvP91W|new++}",
    "Remove {--%2026-06-16T17:35:35+0200|old--}",
    "Discuss <!-- @agent note -->",
  ].join("\n");

  const annotations = collectReviewAnnotations(source);

  assert.deepEqual(
    annotations.map(({ kind, lineStart, lineEnd }) => ({ kind, lineStart, lineEnd })),
    [
      { kind: "addition", lineStart: 1, lineEnd: 1 },
      { kind: "deletion", lineStart: 2, lineEnd: 2 },
      { kind: "conversation", lineStart: 3, lineEnd: 3 },
    ],
  );

  assert.equal(
    applyReviewAnnotationAction(source, annotations[0], { kind: "apply" }),
    [
      "Add new",
      "Remove {--%2026-06-16T17:35:35+0200|old--}",
      "Discuss <!-- @agent note -->",
    ].join("\n"),
  );

  assert.equal(
    applyReviewAnnotationAction(source, annotations[1], { kind: "cancel" }),
    [
      "Add {++%1WzvP91W|new++}",
      "Remove old",
      "Discuss <!-- @agent note -->",
    ].join("\n"),
  );
});

test("CLI status --short prints compact per-file stats with local today time", () => {
  const cwd = makeTempDir();
  const now = new Date();
  const today = formatLocalIso(now.getFullYear(), now.getMonth() + 1, now.getDate(), 11, 16, 34);
  const conversations = [
    ...Array.from({ length: 6 }, (_, index) => `<!-- @agent open ${index} -->`),
    ...Array.from({ length: 5 }, (_, index) => `<!-- @agent wip ${index} @me -->`),
    ...Array.from({ length: 10 }, (_, index) => `<!-- @agent handled ${index} @me answer -->`),
    "<!-- @agent resolved @me ok -->",
  ];
  const annotations = [
    ...Array.from({ length: 3 }, (_, index) => `{++addition ${index}++}`),
    `{++%${today}|addition latest++}`,
    "{--deletion 1--}",
    "{--deletion 2--}",
    "{~~old 1~>new 1~~}",
    "{~~old 2~>new 2~~}",
    ...Array.from({ length: 7 }, (_, index) => `{==highlight ${index}==}`),
    "{>>comment 1<<}",
    "{>>comment 2<<}",
  ];
  fs.writeFileSync(path.join(cwd, "file.md"), [...conversations, ...annotations, ""].join("\n"), "utf8");
  fs.writeFileSync(path.join(cwd, "agent.md"), `<!-- @agent%${today} open -->\n`, "utf8");
  fs.writeFileSync(path.join(cwd, "human.md"), `<!-- @agent older @%${today} answer -->\n`, "utf8");

  const result = spawnSync(
    process.execPath,
    [path.join(__dirname, "..", "out", "dz-review.js"), "status", "--short", "file.md", "agent.md", "human.md"],
    { cwd, encoding: "utf8" },
  );

  assert.equal(result.status, 0, result.stderr);
  assert.deepEqual(result.stdout.trim().split("\n"), [
    "file.md: 11/22 +4-2~2=7x2 - o 11:16:34",
    "agent.md: 1/1 - a 11:16:34",
    "human.md: 0/1 - @ 11:16:34",
  ]);
});
